import googlemaps import datetime # Initialize Google Maps client with your API key gmaps = googlemaps.Client(key='YOUR_API_KEY') # 1. Geocoding API - Convert address to coordinates def get_coordinates(address): geocode_result = gmaps.geocode(address) if geocode_result: location = geocode_result[0]['geometry']['location'] return location['lat'], location['lng'] return None # Example address address = "1600 Amphitheatre Parkway, Mountain View, CA" latitude, longitude = get_coordinates(address) print(f"Coordinates of {address}: {latitude}, {longitude}") # 2. Directions API - Get directions between two locations origin = "New York, NY" destination = "Los Angeles, CA" now = datetime.datetime.now() directions_result = gmaps.directions(origin, destination, mode="driving", departure_time=now) # Print summary of directions for step in directions_result[0]['legs'][0]['steps']: print(step['html_instructions']) # 3. Distance Matrix API - Get travel time and distance between multiple points origins = ["New York, NY", "Boston, MA"] destinations = ["Los Angeles, CA", "Chicago, IL"] distance_matrix_result = gmaps.distance_matrix(origins, destinations) for row in distance_matrix_result['rows']: for element in row['elements']: print(f"Distance: {element['distance']['text']}, Duration: {element['duration']['text']}") # 4.
